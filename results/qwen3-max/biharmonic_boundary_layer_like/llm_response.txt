import numpy as np
from mpi4py import MPI
from dolfinx import mesh, fem, geometry
from dolfinx.fem import petsc
import ufl
from petsc4py import PETSc

ScalarType = PETSc.ScalarType

def solve(case_spec: dict) -> dict:
    # Parameters
    mesh_resolution = 64
    element_degree = 2
    ksp_type = "cg"
    pc_type = "hypre"
    rtol = 1e-8
    
    comm = MPI.COMM_WORLD
    domain = mesh.create_unit_square(comm, mesh_resolution, mesh_resolution, cell_type=mesh.CellType.triangle)
    
    # Mixed function space for mixed formulation: V1 for u, V2 for v = -Δu
    V1 = fem.functionspace(domain, ("Lagrange", element_degree))
    V2 = fem.functionspace(domain, ("Lagrange", element_degree))
    W = V1 * V2
    
    # Exact solution for boundary conditions and source term
    x = ufl.SpatialCoordinate(domain)
    u_exact_expr = ufl.exp(5*(x[0]-1)) * ufl.sin(ufl.pi*x[1])
    
    # Compute f = Δ²u_exact
    laplacian_u = ufl.div(ufl.grad(u_exact_expr))
    biharmonic_u = ufl.div(ufl.grad(laplacian_u))
    f_expr = biharmonic_u
    
    # Boundary condition for u
    u_D = fem.Function(V1)
    u_D.interpolate(lambda x: np.exp(5*(x[0]-1)) * np.sin(np.pi*x[1]))
    
    # Boundary condition for v = -Δu
    laplacian_u_exact = ufl.div(ufl.grad(u_exact_expr))
    v_D_expr = -laplacian_u_exact
    v_D = fem.Function(V2)
    v_D.interpolate(fem.Expression(v_D_expr, V2.element.interpolation_points))
    
    # Trial and test functions
    (u, v) = ufl.TrialFunctions(W)
    (w1, w2) = ufl.TestFunctions(W)
    
    # Mixed variational formulation:
    # -Δu = v  => ∫∇u·∇w1 dx - ∫v w1 dx = 0
    # -Δv = f  => ∫∇v·∇w2 dx = ∫f w2 dx
    f = fem.Constant(domain, ScalarType(0.0))
    a = (ufl.inner(ufl.grad(u), ufl.grad(w1)) - ufl.inner(v, w1) + 
         ufl.inner(ufl.grad(v), ufl.grad(w2))) * ufl.dx
    L = ufl.inner(f, w2) * ufl.dx
    
    # Apply boundary conditions
    tdim = domain.topology.dim
    fdim = tdim - 1
    boundary_facets = mesh.locate_entities_boundary(domain, fdim, lambda x: np.ones(x.shape[1], dtype=bool))
    
    dofs_u = fem.locate_dofs_topological((W.sub(0), V1), fdim, boundary_facets)
    dofs_v = fem.locate_dofs_topological((W.sub(1), V2), fdim, boundary_facets)
    
    bc_u = fem.dirichletbc(u_D, dofs_u, W.sub(0))
    bc_v = fem.dirichletbc(v_D, dofs_v, W.sub(1))
    bcs = [bc_u, bc_v]
    
    # Set up source term
    f_func = fem.Function(V2)
    f_func.interpolate(fem.Expression(f_expr, V2.element.interpolation_points))
    L = ufl.inner(f_func, w2) * ufl.dx
    
    # Solve the linear system
    problem = petsc.LinearProblem(
        a, L, bcs=bcs,
        petsc_options={
            "ksp_type": ksp_type,
            "pc_type": pc_type,
            "ksp_rtol": rtol,
            "ksp_max_it": 1000
        },
        petsc_options_prefix="biharmonic_"
    )
    w_sol = problem.solve()
    u_sol, v_sol = w_sol.split()
    
    # Evaluate on 50x50 grid
    nx, ny = 50, 50
    x_points = np.linspace(0, 1, nx)
    y_points = np.linspace(0, 1, ny)
    X, Y = np.meshgrid(x_points, y_points, indexing='ij')
    points = np.zeros((3, nx*ny))
    points[0] = X.flatten()
    points[1] = Y.flatten()
    
    # Create bounding box tree and find colliding cells
    bb_tree = geometry.bb_tree(domain, domain.topology.dim)
    cell_candidates = geometry.compute_collisions_points(bb_tree, points.T)
    colliding_cells = geometry.compute_colliding_cells(domain, cell_candidates, points.T)
    
    # Evaluate solution at points
    u_values = np.full((nx*ny,), np.nan)
    points_on_proc = []
    cells_on_proc = []
    eval_map = []
    for i in range(nx*ny):
        links = colliding_cells.links(i)
        if len(links) > 0:
            points_on_proc.append(points.T[i])
            cells_on_proc.append(links[0])
            eval_map.append(i)
    
    if len(points_on_proc) > 0:
        vals = u_sol.eval(np.array(points_on_proc), np.array(cells_on_proc, dtype=np.int32))
        u_values[eval_map] = vals.flatten()
    
    u_grid = u_values.reshape((nx, ny))
    
    return {
        "u": u_grid,
        "solver_info": {
            "mesh_resolution": mesh_resolution,
            "element_degree": element_degree,
            "ksp_type": ksp_type,
            "pc_type": pc_type,
            "rtol": rtol
        }
    }
