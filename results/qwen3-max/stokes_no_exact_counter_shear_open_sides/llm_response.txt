import numpy as np
from mpi4py import MPI
from dolfinx import mesh, fem, geometry
from dolfinx.fem import petsc
import ufl
from petsc4py import PETSc

def solve(case_spec: dict) -> dict:
    # Parameters
    mesh_resolution = 64
    degree_u = 2
    degree_p = 1
    ksp_type = "minres"
    pc_type = "hypre"
    rtol = 1e-10
    
    # Setup
    comm = MPI.COMM_WORLD
    domain = mesh.create_unit_square(comm, mesh_resolution, mesh_resolution, cell_type=mesh.CellType.triangle)
    
    # Function spaces (Taylor-Hood)
    V = fem.functionspace(domain, ("Lagrange", degree_u, (domain.geometry.dim,)))
    Q = fem.functionspace(domain, ("Lagrange", degree_p))
    W = V * Q
    
    # Trial and test functions
    (u, p) = ufl.TrialFunctions(W)
    (v, q) = ufl.TestFunctions(W)
    
    # Source term
    f = ufl.as_vector((0.0, 0.0))
    
    # Bilinear and linear forms
    a = (ufl.inner(ufl.grad(u), ufl.grad(v)) - ufl.div(v) * p + ufl.div(u) * q) * ufl.dx
    L = ufl.inner(f, v) * ufl.dx
    
    # Boundary conditions (u = 0 on all boundaries)
    tdim = domain.topology.dim
    fdim = tdim - 1
    boundary_facets = mesh.locate_entities_boundary(domain, fdim, lambda x: np.ones(x.shape[1], dtype=bool))
    
    # Velocity BCs
    u_bc = fem.Function(V)
    u_bc.x.array[:] = 0.0
    dofs_u = fem.locate_dofs_topological((W.sub(0), V), fdim, boundary_facets)
    bc_u = fem.dirichletbc(u_bc, dofs_u, W.sub(0))
    bcs = [bc_u]
    
    # Pressure nullspace handling: pin pressure at a point
    # Find a single vertex to pin pressure
    p_bc = fem.Function(Q)
    p_bc.x.array[:] = 0.0
    # Pin pressure at (0,0)
    def pin_point(x):
        return np.logical_and(np.isclose(x[0], 0.0), np.isclose(x[1], 0.0))
    p_dofs = fem.locate_dofs_geometrical(Q, pin_point)
    if len(p_dofs) > 0:
        bc_p = fem.dirichletbc(p_bc, p_dofs, Q)
        dofs_p = fem.locate_dofs_topological((W.sub(1), Q), 0, p_dofs)
        bc_p_mixed = fem.dirichletbc(p_bc, dofs_p, W.sub(1))
        bcs.append(bc_p_mixed)
    
    # Create problem and solve
    problem = petsc.LinearProblem(
        a, L, bcs=bcs,
        petsc_options={
            "ksp_type": ksp_type,
            "pc_type": pc_type,
            "ksp_rtol": rtol,
            "ksp_atol": 1e-12,
            "ksp_max_it": 1000
        },
        petsc_options_prefix="stokes_"
    )
    w_sol = problem.solve()
    u_sol, p_sol = w_sol.split()
    
    # Evaluate velocity magnitude on 100x100 grid
    nx, ny = 100, 100
    x = np.linspace(0, 1, nx)
    y = np.linspace(0, 1, ny)
    X, Y = np.meshgrid(x, y, indexing='ij')
    points = np.zeros((3, nx * ny))
    points[0] = X.flatten()
    points[1] = Y.flatten()
    
    # Evaluate velocity components
    u_values = u_sol.eval(points.T, np.full(points.shape[1], 0, dtype=np.int32))
    u_x = u_values[:, 0].reshape((nx, ny))
    u_y = u_values[:, 1].reshape((nx, ny))
    u_magnitude = np.sqrt(u_x**2 + u_y**2)
    
    return {
        "u": u_magnitude,
        "solver_info": {
            "mesh_resolution": mesh_resolution,
            "element_degree": {"velocity": degree_u, "pressure": degree_p},
            "ksp_type": ksp_type,
            "pc_type": pc_type,
            "rtol": rtol
        }
    }
