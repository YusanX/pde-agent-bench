import numpy as np
from mpi4py import MPI
from dolfinx import mesh, fem, geometry
from dolfinx.fem import petsc
import ufl
from petsc4py import PETSc

def solve(case_spec: dict) -> dict:
    # Parameters
    mesh_resolution = 32
    degree_u = 2
    degree_p = 1
    ksp_type = "minres"
    pc_type = "jacobi"
    rtol = 1e-8
    
    # Setup
    comm = MPI.COMM_WORLD
    domain = mesh.create_unit_square(comm, nx=mesh_resolution, ny=mesh_resolution, cell_type=mesh.CellType.triangle)
    
    # Function spaces (Taylor-Hood)
    V = fem.functionspace(domain, ("Lagrange", degree_u, (domain.geometry.dim,)))
    Q = fem.functionspace(domain, ("Lagrange", degree_p))
    W = V * Q
    
    # Trial and test functions
    (u, p) = ufl.TrialFunctions(W)
    (v, q) = ufl.TestFunctions(W)
    
    # Source term
    f = ufl.as_vector((0.0, 0.0))
    
    # Bilinear and linear forms
    a = (ufl.inner(ufl.grad(u), ufl.grad(v)) - ufl.div(v) * p + ufl.div(u) * q) * ufl.dx
    L = ufl.inner(f, v) * ufl.dx
    
    # Boundary conditions (rotating wall: all boundaries have u = [0,0] except top boundary has u = [1,0])
    tdim = domain.topology.dim
    fdim = tdim - 1
    
    # Bottom, left, right boundaries: u = [0,0]
    def bottom_left_right(x):
        return np.logical_or.reduce((
            np.isclose(x[1], 0.0),
            np.isclose(x[0], 0.0),
            np.isclose(x[0], 1.0)
        ))
    
    # Top boundary: u = [1,0]
    def top_boundary(x):
        return np.isclose(x[1], 1.0)
    
    # Locate facets
    bottom_left_right_facets = mesh.locate_entities_boundary(domain, fdim, bottom_left_right)
    top_facets = mesh.locate_entities_boundary(domain, fdim, top_boundary)
    
    # Create boundary condition functions
    u_zero = fem.Function(V)
    u_zero.x.array[:] = 0.0
    u_top = fem.Function(V)
    u_top.interpolate(lambda x: np.vstack((np.ones(x.shape[1]), np.zeros(x.shape[1]))))
    
    # Locate DOFs
    dofs_bottom_left_right = fem.locate_dofs_topological((W.sub(0), V), fdim, bottom_left_right_facets)
    dofs_top = fem.locate_dofs_topological((W.sub(0), V), fdim, top_facets)
    
    # Create Dirichlet BCs
    bc_bottom_left_right = fem.dirichletbc(u_zero, dofs_bottom_left_right, W.sub(0))
    bc_top = fem.dirichletbc(u_top, dofs_top, W.sub(0))
    bcs = [bc_bottom_left_right, bc_top]
    
    # Solve using LinearProblem
    problem = petsc.LinearProblem(
        a, L, bcs=bcs,
        petsc_options={
            "ksp_type": ksp_type,
            "pc_type": pc_type,
            "ksp_rtol": rtol,
            "ksp_monitor": False
        },
        petsc_options_prefix="stokes_"
    )
    w_sol = problem.solve()
    u_sol, p_sol = w_sol.split()
    
    # Evaluate on 100x100 grid
    nx, ny = 100, 100
    x = np.linspace(0, 1, nx)
    y = np.linspace(0, 1, ny)
    X, Y = np.meshgrid(x, y, indexing='ij')
    points = np.zeros((3, nx * ny))
    points[0] = X.flatten()
    points[1] = Y.flatten()
    
    # Create bounding box tree
    bb_tree = geometry.bb_tree(domain, domain.topology.dim)
    cell_candidates = geometry.compute_collisions_points(bb_tree, points.T)
    colliding_cells = geometry.compute_colliding_cells(domain, cell_candidates, points.T)
    
    # Evaluate velocity
    u_values = np.zeros((nx * ny, 2))
    points_on_proc = []
    cells_on_proc = []
    for i in range(nx * ny):
        if len(colliding_cells.links(i)) > 0:
            points_on_proc.append(points[:, i])
            cells_on_proc.append(colliding_cells.links(i)[0])
    
    if len(points_on_proc) > 0:
        u_eval = u_sol.eval(np.array(points_on_proc), np.array(cells_on_proc, dtype=np.int32))
        u_values[:len(points_on_proc)] = u_eval
    
    # Compute velocity magnitude
    u_magnitude = np.sqrt(u_values[:, 0]**2 + u_values[:, 1]**2)
    u_grid = u_magnitude.reshape((nx, ny))
    
    return {
        "u": u_grid,
        "solver_info": {
            "mesh_resolution": mesh_resolution,
            "element_degree": degree_u,
            "ksp_type": ksp_type,
            "pc_type": pc_type,
            "rtol": rtol
        }
    }
