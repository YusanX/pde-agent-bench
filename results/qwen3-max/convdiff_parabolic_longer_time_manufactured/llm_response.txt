```python
import numpy as np
from mpi4py import MPI
from dolfinx import mesh, fem, geometry
import ufl
from petsc4py import PETSc
from dolfinx.fem import petsc

def solve(case_spec: dict) -> dict:
    # Parameters from case_spec or defaults
    mesh_resolution = 64
    element_degree = 1
    dt = 0.02
    t_end = 0.2
    ksp_type = "gmres"
    pc_type = "ilu"
    rtol = 1e-8
    
    # Problem parameters
    eps = 0.05
    beta = [2.0, 1.0]
    
    # Create mesh
    comm = MPI.COMM_WORLD
    domain = mesh.create_unit_square(comm, mesh_resolution, mesh_resolution, cell_type=mesh.CellType.triangle)
    
    # Function space
    V = fem.functionspace(domain, ("Lagrange", element_degree))
    
    # Time parameters
    num_steps = int(t_end / dt)
    
    # Define exact solution for boundary and initial conditions
    x = ufl.SpatialCoordinate(domain)
    t_ufl = fem.Constant(domain, PETSc.ScalarType(0.0))
    u_exact_expr = ufl.exp(-2*t_ufl) * ufl.sin(ufl.pi*x[0]) * ufl.sin(ufl.pi*x[1])
    
    # Initial condition
    u_n = fem.Function(V)
    u_n.interpolate(fem.Expression(u_exact_expr, V.element.interpolation_points))
    
    # Boundary condition (Dirichlet on entire boundary)
    def boundary_marker(x):
        return np.logical_or.reduce([
            np.isclose(x[0], 0.0),
            np.isclose(x[0], 1.0),
            np.isclose(x[1], 0.0),
            np.isclose(x[1], 1.0)
        ])
    
    fdim = domain.topology.dim - 1
    boundary_facets = mesh.locate_entities_boundary(domain, fdim, boundary_marker)
    dofs = fem.locate_dofs_topological(V, fdim, boundary_facets)
    u_bc = fem.Function(V)
    bc = fem.dirichletbc(u_bc, dofs)
    
    # Trial and test functions
    u = ufl.TrialFunction(V)
    v = ufl.TestFunction(V)
    
    # Velocity vector
    beta_vec = ufl.as_vector(beta)
    
    # SUPG stabilization parameter
    h = ufl.CellDiameter(domain)
    Pe = ufl.sqrt(ufl.dot(beta_vec, beta_vec)) * h / (2 * eps)
    tau = ufl.conditional(ufl.gt(Pe, 1), h / (2 * ufl.sqrt(ufl.dot(beta_vec, beta_vec))) * (1 - 1/Pe), 0)
    
    # Bilinear and linear forms with SUPG
    a = (u * v / dt + eps * ufl.dot(ufl.grad(u), ufl.grad(v)) + ufl.dot(beta_vec, ufl.grad(u)) * v) * ufl.dx
    a += tau * (u / dt + ufl.dot(beta_vec, ufl.grad(u))) * ufl.dot(beta_vec, ufl.grad(v)) * ufl.dx
    
    # Source term derived from manufactured solution
    u_t = ufl.diff(u_exact_expr, t_ufl)
    laplacian_u = ufl.div(ufl.grad(u_exact_expr))
    conv_term = ufl.dot(beta_vec, ufl.grad(u_exact_expr))
    f_expr = u_t - eps * laplacian_u + conv_term
    
    L = (u_n / dt * v + f_expr * v) * ufl.dx
    L += tau * f_expr * ufl.dot(beta_vec, ufl.grad(v)) * ufl.dx
    
    # Prepare forms
    a_form = fem.form(a)
    L_form = fem.form(L)
    
    # Assemble matrix
    A = petsc.assemble_matrix(a_form, bcs=[bc])
    A.assemble()
    
    # Create RHS vector
    b = petsc.create_vector(L_form.function_spaces)
    
    # Setup solver
    solver = PETSc.KSP().create(comm)
    solver.setOperators(A)
    solver.setType(ksp_type)
    solver.getPC().setType(pc_type)
    solver.rtol = rtol
    
    # Time-stepping
    u_h = fem.Function(V)
    u_h.x.array[:] = u_n.x.array
    
    for n in range(num_steps):
        # Update time
        t = (n + 1) * dt
        t_ufl.value = t
        
        # Update boundary condition
        u_bc_expr = fem.Expression(u_exact_expr, V.element.interpolation_points)
        u_bc.interpolate(u_bc_expr)
        
        # Update source term
        f_expr_updated = u_t - eps * laplacian_u + conv_term
        L_updated = (u_n / dt * v + f_expr_updated * v) * ufl.dx
        L_updated += tau * f_expr_updated * ufl.dot(beta_vec, ufl.grad(v)) * ufl.dx
        L_form_updated = fem.form(L_updated)
        
        # Assemble RHS
        with b.localForm() as loc:
            loc.set(0)
        petsc.assemble_vector(b, L_form_updated)
        petsc.apply_lifting(b, [a_form], bcs=[[bc]])
        b.ghostUpdate(addv=PETSc.InsertMode.ADD, mode=PETSc.ScatterMode.REVERSE)
        petsc.set_bc(b, [bc])
        
        # Solve
        solver.solve(b, u_h.x.petsc_vec)
        u_h.x.scatter_forward()
        
        # Update previous solution
        u_n.x.array[:] = u_h.x.array[:]
    
    # Evaluate on 50x50 grid
    nx, ny = 50, 50
    x_points = np.linspace(0, 1, nx)
    y_points = np.linspace(0, 1, ny)
    X, Y = np.meshgrid(x_points, y_points, indexing='ij')
    points = np.zeros((3, nx * ny))
    points[0] = X.flatten()
    points[1] = Y.flatten()
    
    # Create bounding box tree
    bb_tree = geometry.bb_tree(domain, domain.topology.dim)
    cells = []
    points_on_proc = []
    for i, point in enumerate(points.T):
        cell_candidates = geometry.compute_collisions_points(bb_tree, point.reshape(1, 3))
        colliding_cells = geometry.compute_colliding_cells(domain, cell_candidates, point.reshape(1, 3))
        if len(colliding_cells.links(0)) > 0:
            cells.append(colliding_cells.links(0)[0])
            points_on_proc.append(point)
    
    u_values = np.full((nx * ny,), np.nan)
    if len(points_on_proc) > 0:
        u_eval = u_h.eval(np.array(points_on_proc), np.array(cells, dtype=np.int32))
        u_values = u_eval.flatten()
    
    u_grid = u_values.reshape((nx, ny))
    
    return {
        "u": u_grid,
        "solver_info": {
            "mesh_resolution": mesh_resolution,
            "element_degree": element_degree,
            "ksp_type": ksp_type,
            "pc_type": pc_type,
            "rtol": rtol
        }
    }
```
